String(headers[colIndex]).trim() : ""; return formatCellByRules(cellValue, headerName, sourceType); }); }); var result = { headers: headers, rows: formattedRows }; return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ error: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } /** * Fungsi Pintar Pemformatan Sesuai Aturan Kolom */ function formatCellByRules(val, headerName, sourceType) { if (val === "" || val === null || val === undefined) { return ""; } var hLower = headerName.toLowerCase(); var strVal = String(val).trim(); // 1. FORMAT RUPIAH (Harga / Biaya) if (hLower.includes("harga") || hLower.includes("biaya")) { var num = Number(String(val).replace(/[^0-9.-]+/g,"")); if (!isNaN(num) && num !== 0) { return "Rp " + num.toLocaleString("id-ID"); } } // 2. FORMAT ONLY CLOCK (Waktu / Jam saja, Bersih dari 1899 & GMT) if (hLower.includes("waktu") || hLower.includes("jam")) { // Jika dari objek Date atau string waktu var d = parseToDate(val); if (d) { var hh = String(d.getHours()).padStart(2, '0'); var mm = String(d.getMinutes()).padStart(2, '0'); // Jika jam/menit valid (bukan 00:00 default mutlak akibat salah baca 1899) if (hh !== "00" || mm !== "00" || strVal.includes("00:00")) { // Cek apakah aslinya memang jam 00.00 atau karena parsing 1899 if (!strVal.includes("1899-12-30 00:00")) { return hh + "." + mm; } } } // Ekstraksi teks manual jika string mengandung format jam if (strVal.includes("1899") || strVal.includes("T")) { var timeMatch = strVal.match(/(\d{2}):(\d{2})/); if (timeMatch && timeMatch[1] !== "00") { return timeMatch[1] + "." + timeMatch[2]; } } // Jika bentuknya string jam biasa misal "09:07:00" var simpleTime = strVal.match(/^(\d{2}):(\d{2})/); if (simpleTime) { return simpleTime[1] + "." + simpleTime[2]; } } // 3. FORMAT ONLY DATE (Tanggal saja, tanpa jam 00.00) if (hLower.includes("tanggal") || hLower.includes("tgl")) { var dDate = parseToDate(val); if (dDate) { return formatDateID(dDate); } } // 4. FORMAT DATE AND CLOCK (Timestamp / Waktu Lengkap) if (hLower.includes("timestamp")) { var dFull = parseToDate(val); if (dFull) { return formatDateID(dFull) + ", " + String(dFull.getHours()).padStart(2, '0') + "." + String(dFull.getMinutes()).padStart(2, '0'); } } // 5. FORMAT MONTH (Periode Bulan) if (hLower.includes("periode")) { var dMonth = parseToDate(val); if (dMonth) { var monthsFull = ["Januari", "Februari", "Maret", "April", "Mei", "Juni", "Juli", "Agustus", "September", "Oktober", "November", "Desember"]; return monthsFull[dMonth.getMonth()] + " " + dMonth.getFullYear(); } } // Fallback umum untuk data tanggal/waktu lain yang lolos if (val instanceof Date || (strVal.match(/^\d{4}-\d{2}-\d{2}/) && !hLower.includes("nomor"))) { var dGen = parseToDate(val); if (dGen) { return formatDateID(dGen); } } return strVal; } function parseToDate(val) { if (val instanceof Date) return val; var parsed = new Date(val); if (!isNaN(parsed.getTime())) return parsed; return null; } function formatDateID(dateObj) { var d = String(dateObj.getDate()).padStart(2, '0'); var months = ["Jan", "Feb", "Mar", "Apr", "Mei", "Jun", "Jul", "Ags", "Sep", "Okt", "Nov", "Des"]; var m = months[dateObj.getMonth()]; var y = dateObj.getFullYear(); return d + " " + m + " " + y; }
